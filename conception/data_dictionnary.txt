
DATA DICTIONNARY

Users :
id - INT PK NOT NULL
password - TEXT NOT NULL
mail - TEXT UNIQUE NOT NULL (Registration email)
pseudo - TEXT (Pseudo displayed to other users)
settings_theme - TEXT (App theme)
settings_volume - INT (App global volume)
created_at - DATE NOT NULL
updated_at - DATE
favorite_sounds - ASSOCIATION (Users_Sounds)
favorite_ambiances - ASSOCIATION (Users_Ambiances)

Sounds :
id - INT PK NOT NULL
name - TEXT UNIQUE NOT NULL (Displayed name)
audio_paths - TEXT[] NOT NULL (Audio paths for a sound, a sound can play a random audio from these paths)
image_path - TEXT NOT NULL
looping - BOOLEAN NOT NULL (Make a sound loop)
volume - INT[] NOT NULL (Set minimum and maximum volume)
reverb - INT[] NOT NULL (Set minimum and maximum reverb)
direction - INT[] NOT NULL (Set left and right panning)
repeat_delay - DECIMAL[] NOT NULL (Set minimum and maximum delay between played audios)
category - TEXT (Vegetation, animals, insects, human, objects etc)

Ambiances :
id - INT PK NOT NULL
name - TEXT NOT NULL
author_id - TEXT NOT NULL (The user who created the ambiance)
likes - INT (How many times the ambiance has been added to favorites)
data - JSONB (Settings of each section and sound of the ambiance)
total_duration - INT (Sum of the duration of each section)
sections - INT (Number of sections)
used_sounds - ASSOCIATION (Table Ambiances_Sounds)

Plans :
id - INT PK NOT NULL
name - TEXT NOT NULL (Free, premium, enterprise)
price - DECIMAL NOT NULL
description - TEXT

Subscriptions :
id - INT PK NOT NULL
user_id - INT FK NOT NULL (User that subscribed to the plan)
plan_id - INT FK NOT NULL (Which plan has been chosen)
starting_date - DATE NOT NULL (Starting date of the subscription)
ending_date - DATE NOT NULL (Ending date of the subscription)
payment_date - DATE NOT NULL (Date of payment reception)
created_at - DATE NOT NULL
status - ENUM(TEXT) (Status of the subscription, like active, concelled or expired)

Ambiances_Sounds
id - INT PK NOT NULL
ambiance_id - INT FK NOT NULL
sound_id - INT FK NOT NULL

Users_Sounds :
id - INT PK NOT NULL
user_id - INT FK NOT NULL
sound_id - INT FK NOT NULL

Users_Ambiances :
id - INT PK NOT NULL
user_id - INT FK NOT NULL
ambiance_id - INT FK NOT NULL
